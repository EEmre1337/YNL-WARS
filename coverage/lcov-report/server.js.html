import express from 'express';
import { createServer } from 'http';
import { Server } from 'socket.io';
 
const app = express();
const server = createServer(app);
const io = new Server(server);
 
const PORT = 3000;
 
app.use(express.static('public'));
 
// ========================
// Spiellogik Variablen
// ========================
const players = {};
const projectiles = [];
const walls = [
    { x: 300, y: 200, width: 10, height: 400 },
    { x: 1200 -10, y: 200, width: 10, height: 400 },
    { x: 700, y: 350, width: 10, height: 100 },
    { x: 800 -10, y: 350, width: 10, height: 100 },
    { x: 700, y: 200, width: 100, height: 10 },
    { x: 700, y: 600 -10 , width: 100, height: 10 }
];
 
const redSpawn = { x: 100, y: 400, width: 200, height: 200 };
const blueSpawn = { x: 1400, y: 400, width: 200, height: 200 };
const flag = { x: 750, y: 400, holder: null };
const score = { red: 0, blue: 0 };
const border = { x: 0, y: 0, width: 1500, height: 800 };
const chatMessages = [];
 
// ========================
// Hilfsfunktionen
// ========================
function checkCollision(x, y, width, height, walls) {
    return walls.some(wall =>
        x < wall.x + wall.width &&
        x + width > wall.x &&
        y < wall.y + wall.height &&
        y + height > wall.y
    );
}
 
function checkBorderCollision(x, y, width, height) {
    return x >= border.x && x <= border.x + border.width - width && y >= border.y && y <= border.y + border.height - height;
}
 
// ========================
// Socket.io Events
// ========================
io.on('connection', (socket) => {
    console.log('Neuer Spieler verbunden:', socket.id);
    socket.emit('chatUpdate', chatMessages);

    socket.on('chooseTeam', (data) => {
        console.log("üì¢ Empfangen: ", data);
    
        const { username, team } = data || {};
    
        if (!username || !team) {
            console.error("‚ùå Fehler: `username` oder `team` ist undefined!", { username, team });
            return;
        }
    
        players[socket.id] = {
            x: team === 'red' ? redSpawn.x : blueSpawn.x,
            y: team === 'red' ? redSpawn.y : blueSpawn.y,
            team,
            username,
            hasFlag: false
        };
    
        console.log(`üì¢ ${username} ist Team ${team.toUpperCase()} beigetreten!`);
        chatMessages.push(`üì¢ ${username} ist Team ${team.toUpperCase()} beigetreten!`);
        io.emit('chatUpdate', chatMessages);
    
        // **Spielstatus sofort an den Client senden**
        io.emit('state', { players, projectiles, walls, flag, score, redSpawn, blueSpawn });
    });
    

socket.on('move', (data) => {
    const player = players[socket.id];
    if (!player) return;

    const newX = player.x + data.dx;
    const newY = player.y + data.dy;

    // √úberpr√ºfe, ob der Spieler nicht in eine Wand oder aus der Grenze l√§uft
    if (!checkCollision(newX - 10, newY - 10, 20, 20, walls) && checkBorderCollision(newX, newY, 20, 20)) {
        player.x = newX;
        player.y = newY;
    }

    // **üîπ Flagge aufnehmen**
    if (!flag.holder && Math.abs(player.x - flag.x) < 15 && Math.abs(player.y - flag.y) < 15) {
        flag.holder = socket.id;
        player.hasFlag = true;
        console.log(`${player.username} hat die Flagge aufgenommen!`);
        io.emit('chatMessage', `${player.username} hat die Flagge aufgenommen!`);
    }

    // **üèÅ Flagge ins Ziel bringen**
    const spawn = player.team === 'red' ? redSpawn : blueSpawn;
    if (player.hasFlag && player.x > spawn.x && player.x < spawn.x + spawn.width && player.y > spawn.y && player.y < spawn.y + spawn.height) {
        score[player.team]++;
        flag.x = 750; // Flagge zur√ºcksetzen
        flag.y = 400;
        flag.holder = null;
        player.hasFlag = false;

        io.emit('chatMessage', `${player.username} hat einen Punkt f√ºr Team ${player.team.toUpperCase()} erzielt!`);
        console.log(`üèÜ Team ${player.team} hat jetzt ${score[player.team]} Punkte`);

        if (score[player.team] === 3) {
            io.emit('gameOver', player.team);
        }
    }
});

socket.on('chatMessage', (msg) => {
    if (!players[socket.id]) return; // Nachricht ignorieren, falls Spieler nicht existiert

    const message = `${players[socket.id].username}: ${msg}`;
    console.log("üì© Chat-Nachricht erhalten:", message);
    
    chatMessages.push(message);
    io.emit('chatUpdate', chatMessages);
});

    socket.on('shoot', (data) => {
        const player = players[socket.id];
        if (!player) return;
        projectiles.push({
            x: player.x,
            y: player.y,
            vx: Math.cos(data.angle) * 5,
            vy: Math.sin(data.angle) * 5,
            owner: socket.id
        });
    });

    socket.on('disconnect', () => {
        if (players[socket.id]) {
            const leaveMessage = `‚ùå ${players[socket.id].username} hat das Spiel verlassen.`;
            chatMessages.push(leaveMessage);
            io.emit('chatUpdate', chatMessages);
            delete players[socket.id];
        }
        console.log('Spieler getrennt:', socket.id);
    });
});
 
const projectileUpdateInterval = setInterval(() => {
    projectiles.forEach((proj, index) => {
        proj.x += proj.vx;
        proj.y += proj.vy;

        if (checkCollision(proj.x - 5, proj.y - 5, 10, 10, walls)) {
            projectiles.splice(index, 1);
            return;
        }

        for (const id in players) {
            const player = players[id];
            if (proj.owner !== id && Math.abs(proj.x - player.x) < 10 && Math.abs(proj.y - player.y) < 10) {
                console.log(`${player.username} wurde eliminiert!`);
                io.emit('chatMessage', `${player.username} wurde eliminiert!`);

                player.x = player.team === 'red' ? redSpawn.x + 50 : blueSpawn.x + 50;
                player.y = player.team === 'red' ? redSpawn.y + 50 : blueSpawn.y + 50;

                if (player.hasFlag) {
                    flag.x = proj.x;
                    flag.y = proj.y;
                    flag.holder = null;
                    player.hasFlag = false;
                }

                projectiles.splice(index, 1);
                break;
            }
        }
    });

    io.emit('state', { players, projectiles, walls, flag, score, redSpawn, blueSpawn });
}, 1000 / 60);
